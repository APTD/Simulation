60  @Override
61  public final List<DoubleMatrix1D> route( final ObjectMatrix2D p_objects, final DoubleMatrix1D p_currentposition, final DoubleMatrix1D p_targetposition )
62  {
63  final TreeSet<CJumpPoint> l_openlist = new TreeSet<>( new CCompareJumpPoint() );
64  final ArrayList<DoubleMatrix1D> l_closedlist = new ArrayList<>();
65  final List<DoubleMatrix1D> l_finalpath = new ArrayList<>();
66 
67  l_openlist.add( new CJumpPoint( p_currentposition, null ) );
68  while ( !l_openlist.isEmpty() )
69  {
70  final CJumpPoint l_currentnode = l_openlist.pollFirst();
71  //if the current node is the end node
72  if ( l_currentnode.coordinate().equals( p_targetposition ) )
73  {
74  l_finalpath.add( p_targetposition );
75  CJumpPoint l_parent = l_currentnode.parent();
76  while ( !l_parent.coordinate().equals( p_currentposition ) )
77  {
78  l_finalpath.add( l_parent.coordinate() );
79  l_parent = l_parent.parent();
80  }
81  Collections.reverse( l_finalpath );
82  return l_finalpath;
83  }
84  //Find the successors to current node (add them to the open list)
85  this.successors( p_objects, l_currentnode, p_targetposition, l_closedlist, l_openlist );
86 
87  //Add the l_currentnode to the closed list (as to not open it again)
88  l_closedlist.add( l_currentnode.coordinate() );
89 
90  }
91 
92  return Collections.<DoubleMatrix1D>emptyList();
93  }

95  @Override
96  public final double estimatedtime( final Stream<DoubleMatrix1D> p_route, final double p_speed )
97  {
98  return StreamUtils.windowed( p_route, 2, 1 )
99  .mapToDouble( i -> Math.sqrt(
100  Algebra.DEFAULT.norm2(
101  new DenseDoubleMatrix1D( i.get( 1 ).toArray() )
102  .assign( i.get( 0 ), Functions.minus )
103  )
104  ) / p_speed
105  )
106  .sum();
107  }

117  private void successors( final ObjectMatrix2D p_objects, final CJumpPoint p_curnode, final DoubleMatrix1D p_target, final ArrayList<DoubleMatrix1D> p_closedlist,
118  final Set<CJumpPoint> p_openlist )
119  {
120  IntStream.rangeClosed( -1, 1 )
121  .forEach( i -> IntStream.rangeClosed( -1, 1 )
122  .filter( j -> ( i != 0 || j != 0 )
123  && !this.isNotNeighbour(
124  p_objects,
125  p_curnode.coordinate().getQuick( 0 ) + i,
126  p_curnode.coordinate().getQuick( 1 ) + j,
127  p_closedlist
128  )
129  && !this.isOccupied(
130  p_objects,
131  p_curnode.coordinate().getQuick( 0 ) + i,
132  p_curnode.coordinate().getQuick( 1 ) + j )
133  )
134  .forEach( j ->
135  this.addsuccessors(
136  this.jump( p_curnode.coordinate(), p_target, i, j, p_objects ),
137  p_closedlist, p_openlist, p_curnode, p_target
138  )
139  ) );
140  }

150  private void addsuccessors( final DoubleMatrix1D p_nextjumpnode, final ArrayList<DoubleMatrix1D> p_closedlist,
151  final Set<CJumpPoint> p_openlist, final CJumpPoint p_curnode, final DoubleMatrix1D p_target )
152  {
153  if ( !( p_nextjumpnode != null && !p_curnode.coordinate().equals( p_nextjumpnode ) && ( !p_closedlist.contains( p_nextjumpnode ) ) ) )
154  return;
155 
156  final CJumpPoint l_jumpnode = new CJumpPoint( p_nextjumpnode, p_curnode );
157  this.calculateScore( l_jumpnode, p_target );
158  p_openlist.removeIf( s -> s.coordinate().equals( p_nextjumpnode ) && s.fscore() > l_jumpnode.fscore() );
159 
160  //checking that the jump point is already exists in open list or not, if yes then check their fscore to make decision
161  if ( !p_openlist.parallelStream().filter( s -> s.coordinate().equals( p_nextjumpnode ) ).anyMatch( s -> s.fscore() < l_jumpnode.fscore() ) )
162  p_openlist.add( l_jumpnode );
163 
164  }

175  private DoubleMatrix1D jump( final DoubleMatrix1D p_curnode, final DoubleMatrix1D p_target,
176  final double p_row, final double p_col, final ObjectMatrix2D p_objects )
177  {
178  //The next nodes details
179  final double l_nextrow = p_curnode.getQuick( 0 ) + p_row;
180  final double l_nextcol = p_curnode.getQuick( 1 ) + p_col;
181  final DoubleMatrix1D l_nextnode = new DenseDoubleMatrix1D( new double[]{l_nextrow, l_nextcol} );
182 
183  // If the l_nextnode is outside the grid or occupied then return null
184  if ( this.isOccupied( p_objects, l_nextrow, l_nextcol ) || this.isNotCoordinate( p_objects, l_nextrow, l_nextcol ) )
185  return null;
186 
187  //If the l_nextnode is the target node then return it
188  if ( p_target.equals( l_nextnode ) )
189  return l_nextnode;
190 
191  //If we are going in a diagonal direction check for forced neighbors
192  if ( p_row != 0 && p_col != 0 )
193  {
194  final DoubleMatrix1D l_node = this.diagjump( l_nextrow, l_nextcol, l_nextnode, p_target, p_row, p_col, p_objects );
195  if ( l_node != null )
196  return l_node;
197  }
198  else
199  if ( this.horizontal( l_nextrow, l_nextcol, p_row, p_objects, 1 ) || this.horizontal( l_nextrow, l_nextcol, p_row, p_objects, -1 )
200  || this.vertical( l_nextrow, l_nextcol, p_col, p_objects, 1 ) || this.vertical( l_nextrow, l_nextcol, p_col, p_objects, -1 ) )
201  return l_nextnode;
202 
203  return this.jump( l_nextnode, p_target, p_row, p_col, p_objects );
204  }

217  private DoubleMatrix1D diagjump( final double p_nextrow, final double p_nextcolumn, final DoubleMatrix1D p_nextnode, final DoubleMatrix1D p_target,
218  final double p_row, final double p_col, final ObjectMatrix2D p_objects )
219  {
220  if ( this.diagonal( p_nextrow, p_nextcolumn, -p_row, p_col, p_row, 0, p_objects ) || this.diagonal( p_nextrow, p_nextcolumn, p_row, -p_col, 0, p_col, p_objects ) )
221  return p_nextnode;
222 
223  //before each diagonal step the algorithm must first fail to detect any straight jump points
224  if ( this.jump( p_nextnode, p_target, p_row, 0, p_objects ) != null || this.jump( p_nextnode, p_target, 0, p_col, p_objects ) != null )
225  return p_nextnode;
226 
227  return null;
228  }

242  final ObjectMatrix2D p_objects )
243  {
244  return !this.isNotCoordinate( p_objects, p_nextrow - p_hzon, p_nextcolumn - p_ver ) && !this.isNotCoordinate( p_objects, p_nextrow + p_row, p_nextcolumn + p_col )
245  && this.isOccupied( p_objects, p_nextrow - p_hzon, p_nextcolumn - p_ver ) && !this.isOccupied( p_objects, p_nextrow + p_row, p_nextcolumn + p_col );

321  private void calculateScore( final CJumpPoint p_jumpnode, final DoubleMatrix1D p_target )
322  {
323  final double l_hscore = Math.abs( p_target.getQuick( 0 ) - p_jumpnode.coordinate().getQuick( 0 ) ) * 10
324  + Math.abs( p_target.getQuick( 1 ) - p_jumpnode.coordinate().getQuick( 1 ) ) * 10;
325 
326  p_jumpnode.hscore( l_hscore );
327 
328  final double l_row = Math.abs( p_jumpnode.parent().coordinate().getQuick( 0 ) - p_jumpnode.coordinate().getQuick( 0 ) );
329  final double l_column = Math.abs( p_jumpnode.parent().coordinate().getQuick( 1 ) - p_jumpnode.coordinate().getQuick( 1 ) );
330  final double l_gscore = ( l_row != 0 && l_column != 0 ) ? Math.abs( 14 * l_row ) : Math.abs( 10 * Math.max( l_row, l_column ) );
331 
332  p_jumpnode.gscore( p_jumpnode.parent().gscore() + l_gscore );
333  }
