127  protected synchronized Instant determinenextstatechange()
128  {
129  if ( m_passengersgenerated >= m_passengers ) return Instant.MAX;
130  return Instant.ofEpochMilli( m_startmillis + (long) Math.ceil( m_distribution.inverseCumulativeProbability(
131  1.0 * ( m_passengersgenerated + 1 ) / m_passengers ) ) );
132  }
