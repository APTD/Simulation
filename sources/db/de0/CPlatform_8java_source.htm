110  @Override
111  protected boolean updatestate()
112  {
113  if ( m_input.isEmpty() ) return false;
114 
115  final List<IMessage> l_arrivingtrains = m_input.get( EMessageType.TRAIN_TO_PLATFORM_ARRIVING );
116  final List<IMessage> l_departingtrains = m_input.get( EMessageType.TRAIN_TO_PLATFORM_DEPARTING );
117  final List<IMessage> l_subscribingpassengers = m_input.get( EMessageType.PASSENGER_TO_PLATFORM_SUBSCRIBE );
118  final List<IMessage> l_unsubscribingpassengers = m_input.get( EMessageType.PASSENGER_TO_PLATFORM_UNSUBSCRIBE );
119 
120  if ( l_departingtrains.size() > 1 ) throw new RuntimeException( m_id + " has multiple trains departing simultaneously at " + m_time.current() );
121  if ( l_departingtrains.size() == 1 )
122  {
123  if ( m_train != l_departingtrains.get( 0 ).sender() )
124  throw new RuntimeException( m_id + " has a train departing that's not there at " + m_time.current() );
125  m_passengers.stream().forEach( p -> output( new CMessage( this, p.id(), EMessageType.PLATFORM_TO_PASSENGER_TRAINDEPARTED, m_train.id() ) ) );
126  m_train = null;
127  m_doors.clear();
128  }
129 
130  l_subscribingpassengers.stream().forEach( msg ->
131  {
132  m_passengers.add( (IPassenger) msg.sender() );
133  if ( m_train != null ) output( new CMessage( this, msg.sender().id(), EMessageType.PLATFORM_TO_PASSENGER_TRAINARRIVED, m_train.id(),
134  m_doors.toArray() ) );
135  } );
136  l_unsubscribingpassengers.stream().forEach( msg -> m_passengers.remove( msg.sender() ) );
137 
138  if ( l_arrivingtrains.size() > 1 ) throw new RuntimeException( m_id + " has multiple trains arriving simultaneously at " + m_time.current() );
139  if ( l_arrivingtrains.size() == 1 )
140  {
141  if ( m_train != null )
142  throw new RuntimeException( m_id + " has a second train arriving without the first departing at " + m_time.current() );
143  m_train = (ITrain) l_arrivingtrains.get( 0 ).sender();
144  Arrays.stream( l_arrivingtrains.get( 0 ).content() ).map( o -> (IDoor<?>) o ).sorted( Comparator.comparing( IDoor::id ) )
145  .forEachOrdered( m_doors::add );
146  m_passengers.stream().forEach( p -> output( new CMessage( this, p.id(), EMessageType.PLATFORM_TO_PASSENGER_TRAINARRIVED, m_train.id(),
147  m_doors.toArray() ) ) );
148  }
149  return true;
150  }
