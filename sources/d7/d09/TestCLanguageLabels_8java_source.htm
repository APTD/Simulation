179  final Set<String> l_ignoredlabel = new HashSet<>();
180

182  final Set<String> l_label = Collections.unmodifiableSet(
183  Files.walk( Paths.get( SEARCHPATH ) )
184  .filter( Files::isRegularFile )
185  .filter( i -> i.toString().endsWith( ".java" ) )
186  .flatMap( i ->
187  {
188  try

192  return l_parser.labels().stream();
193  }
194  catch ( final IOException l_excpetion )
195  {
196  assertTrue( MessageFormat.format( "io error on file [{0}]: {1}", i, l_excpetion.getMessage() ), false );
197  return Stream.empty();
198  }
199  catch ( final ParseProblemException l_exception )
200  {
201  // add label build by class path to the ignore list
202  l_ignoredlabel.add(
203  i.toAbsolutePath().toString()
204  // remove path to class directory
205  .replace(
206  FileSystems.getDefault()
207  .provider()
208  .getPath( SEARCHPATH )
209  .toAbsolutePath()
210  .toString(),
211  ""
212  )
213  // string starts with path separator
214  .substring( 1 )
215  // remove file extension
216  .replace( ".java", "" )
217  // replace separators with dots
218  .replace( "/", CLASSSEPARATOR )
219  // convert to lower-case
220  .toLowerCase()
221  // remove package-root name
222  .replace( CCommon.PACKAGEROOT + CLASSSEPARATOR, "" )
223  );
224 
225  System.err.println( MessageFormat.format( "parsing error on file [{0}]:\n{1}", i, l_exception.getMessage() ) );
226  return Stream.empty();
227  }
228  } )
229  .collect( Collectors.toSet() )
230  );
231

235  // --- check label towards the property definition

390  private String label( final String p_line )
391  {
392  final Matcher l_matcher = LANGUAGEMETHODPATTERN.matcher( p_line );
393  if ( !l_matcher.find() )
394  return "";
395 
396  final String[] l_split = l_matcher.group( 0 ).split( "," );
397  final String[] l_return = new String[2];
398 
399  // class name
400  l_return[0] = l_split[0].replace( TRANSLATEMETHODNAME, "" ).replace( "(", "" ).trim();
401  // label name
402  l_return[1] = l_split[1].replace( ")", "" ).replace( "\"", "" ).split( ";" )[0].trim().toLowerCase( Locale.ROOT );
403 
404  return (
405  "this".equals( l_return[0] )
406  ? buildlabel( m_package, m_outerclass, m_outerclass, m_innerclass, l_return[1] )
407  : buildlabel( m_package, m_outerclass, l_return[0].replace( ".class", "" ).replace( m_package + CLASSSEPARATOR, "" ), "", l_return[1] )
408  ).trim().toLowerCase( Locale.ROOT ).replace( CCommon.PACKAGEROOT + CLASSSEPARATOR, "" );
409 
410  }
