234  final IFactory p_factory, final ITime p_time, final Map<String, IPlatform<?>> p_platforms )

237  final Set<IAction> l_actions = CCommon.actionsFromPackage().collect( Collectors.toSet() );
238  final ListMultimap<String, IPlatform<?>> l_platformbystationid = Multimaps.index( p_platforms.values(), IPlatform::stationid );
239  return Collections.<String, IStation<?>>unmodifiableMap(
240  p_network.getInfrastructure()

245  .map( i -> agentname( i, i.getAny() ) )
246  .map( i -> l_generators.computeIfAbsent(
247  i.getRight(),

249  ).generatesingle(
250  i.getLeft().getId(),
251  i.getLeft().getGeoCoord().getCoord().get( 0 ),

253  l_platformbystationid.get( i.getLeft().getId() )

358  private static Pair<Map<String, ITrain<?>>, Map<String, IDoor<?>>> train( final Network p_network, final Map<String, String> p_agents,
359  final IFactory p_factory, final ITime p_time )
360  {
361  final String l_dooragent = IStatefulElement.getDefaultAsl( "door" );
362  final Map<String, IElement.IGenerator<ITrain<?>>> l_generators = new ConcurrentHashMap<>();
363  final Set<IAction> l_actions = CCommon.actionsFromPackage().collect( Collectors.toSet() );
364  final IElement.IGenerator<IDoor<?>> l_doorgenerator = doorgenerator( p_factory, l_dooragent, l_actions, p_time );
365  final Map<String, AtomicLong> l_doorcount = Collections.synchronizedMap( new HashMap<>() );
366  final Map<String, IDoor<?>> l_doors = Collections.synchronizedMap( new HashMap<>() );
367  return new ImmutablePair<>( Collections.<String, ITrain<?>>unmodifiableMap(
368  p_network.getTimetable()
369  .getTrains()
370  .getTrain()
371  .parallelStream()
372  .filter( i -> hasagentname( i.getAny3() ) )
373  .map( i -> agentname( i, i.getAny3() ) )
374  .map( i -> l_generators.computeIfAbsent(
375  i.getRight(),
376  a -> traingenerator( p_factory, p_agents.get( i.getRight() ), l_actions, p_time )
377  ).generatesingle(
378  i.getLeft().getId(),
379  i.getLeft()
380  .getTrainPartSequence()
381  .stream()
382  .flatMap( ref ->
383  {
384  // @todo support multiple train parts
385  final EOcpTT[] l_tts = ( (ETrainPart) ref.getTrainPartRef().get( 0 ).getRef() )
386  .getOcpsTT().getOcpTT().toArray( new EOcpTT[0] );
387  final CTrain.CTimetableEntry[] l_entries = new CTrain.CTimetableEntry[l_tts.length];
388  for ( int j = 0; j < l_tts.length; j++ )
389  {
390  final EArrivalDepartureTimes l_times = l_tts[j].getTimes().stream()
391  .filter( t -> t.getScope().equalsIgnoreCase( "published" ) )
392  .findAny()
393  .orElseThrow( () -> new CSemanticException( "missing published times" ) );
394  l_entries[j] = new CTrain.CTimetableEntry(
395  j < 1 ? 0.0 : ( (ETrack) l_tts[j - 1].getSectionTT().getTrackRef().get( 0 ).getRef() ).getTrackTopology()
396  .getTrackEnd().getPos()
397  .doubleValue(),
398  ( (EOcp) l_tts[j].getOcpRef() ).getId(),
399  l_tts[j].getStopDescription().getOtherAttributes().getOrDefault( PLATFORM_REF_ATTRIBUTE, null ),
400  l_times.getArrival() == null
401  ? null
402  : l_times.getArrival().toGregorianCalendar().toZonedDateTime()
403  .with( LocalDate.from( p_time.current().atZone( ZoneId.systemDefault() ) ) ).toInstant(),
404  l_times.getDeparture() == null
405  ? null
406  : l_times.getDeparture().toGregorianCalendar().toZonedDateTime()
407  .with( LocalDate.from( p_time.current().atZone( ZoneId.systemDefault() ) ) ).toInstant()
408  );
409  }
410  return Arrays.stream( l_entries );
411  } ),
412  i.getLeft()
413  .getTrainPartSequence()
414  .stream()
415  // @todo support multiple train parts
416  .map( s -> (ETrainPart) s.getTrainPartRef().get( 0 ).getRef() )
417  .map( p -> (EFormation) p.getFormationTT().getFormationRef() )
418  .flatMap( f -> f.getTrainOrder().getVehicleRef().stream() )
419  .map( r -> new ImmutablePair<BigInteger, TDoors>( r.getVehicleCount(),
420  ( (EVehicle) r.getVehicleRef() ).getWagon().getPassenger().getDoors() ) )
421  .flatMap( v -> IntStream.range( 0, v.getLeft().intValue() * v.getRight().getNumber().intValue() )
422  .mapToObj( j -> l_doors.computeIfAbsent( "door-" + i.getLeft().getId() + "-"
423  + l_doorcount.computeIfAbsent(
424  i.getLeft().getId(),
425  id -> new AtomicLong( 1L ) ).getAndIncrement(),
426  id -> l_doorgenerator.generatesingle( id, i.getLeft().getId(),
427  v.getRight().getEntranceWidth().doubleValue()
428  / v.getRight().getNumber().longValue() )
429  )
430  )
431  )
432  .collect( Collectors.toList() )
433  )
434  )
435  .collect( Collectors.toMap( IElement::id, i -> i ) )
436  ), l_doors );
437  }
