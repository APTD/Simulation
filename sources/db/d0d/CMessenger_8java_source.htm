48  @Override
49  public IMessenger call()
50  {
51  if ( m_experiment == null )
52  {
53  java.util.logging.Logger.getLogger( this.getClass().getCanonicalName() ).warning( "CMessenger executed without reference to IExperiment, cannot operate" );
54  return this;
55  }
56  optionalfilteredstream( m_experiment.objects().parallel(), OUTPUT_ONLY_FROM_IMMINENT ? IElement::imminent : null )
57  .flatMap( IElement::output )
58  .peek( msg -> Logger.debug( msg.type() + " from " + msg.sender().id() + " to " + msg.recipient() ) )
59  .forEach( msg -> m_experiment.getAgent( msg.recipient() ).input( msg ) );
60  return this;
61  }
