154  @Override
155  protected Instant determinenextstatechange()
156  {
157  switch ( m_state )
158  {
159  case MOVING_THROUGH_STATION:
160  return m_lastcontinuousupdate.plus(
161  Math.round( ( m_distancetonextplatform - m_distancewalked ) / m_speedatstation ),
162  ChronoUnit.SECONDS
163  );
164  case ENTERING_TRAIN:
165  return m_lastcontinuousupdate.plus(
166  Math.round( m_entranceduration - m_dooruse ),
167  ChronoUnit.SECONDS
168  );
169  case LEAVING_TRAIN:
170  return m_lastcontinuousupdate.plus(
171  Math.round( m_exitduration - m_dooruse ),
172  ChronoUnit.SECONDS
173  );
174  default:
175  return Instant.MAX;
176  }
177  }

179  @Override
180  protected boolean updatestate()
181  {
182  if ( m_input.size() > 1 ) throw new RuntimeException( m_id + " received multiple input messages at once, not possible" );
183  // this may have to be refined if further state logic is added
184 
185  if ( handleplatformmessages() || handledoormessage() ) return true;
186 
187  final List<IMessage> l_trainarriving = m_input.get( EMessageType.TRAIN_TO_PASSENGER_ARRIVING );
188  if ( !l_trainarriving.isEmpty() )
189  {
190  if ( m_state != EPassengerState.ON_TRAIN ) throw new RuntimeException( m_id + " received ARRIVING although in state " + m_state );
191  if ( !m_itinerary.get( m_itindex ).m_arrivalstation.equals( (String) l_trainarriving.get( 0 ).content()[0] ) ) return false;
192  final IDoor<?> l_door = (IDoor<?>) ( (Object[]) l_trainarriving.get( 0 ).content()[2] )[0];
193  output( new CMessage( this, l_door.id(), EMessageType.PASSENGER_TO_DOOR_ENQUEUE_EXIT ) );
194  m_state = EPassengerState.IN_EXIT_QUEUE;
195  return true;
196  }
197 
198  final boolean l_timedchange = !m_nextstatechange.isAfter( m_time.current() );
199  if ( l_timedchange )
200  {
201  switch ( m_state )
202  {
203  case LEAVING_TRAIN:
204  output( new CMessage( this, m_door.id(), EMessageType.PASSENGER_TO_DOOR_FINISHED ) );
205  m_door = null;
206  if ( m_itindex + 1 >= m_itinerary.size() )
207  {
208  System.out.println( m_id + " finished itinerary :-) at " + m_time.current() );
209  m_state = EPassengerState.IDLE_AT_STATION;
210  }
211  else
212  {
213  m_itindex++;
214  m_distancewalked = 0.0;
215  m_state = EPassengerState.MOVING_THROUGH_STATION;
216  }
217  break;
218  case MOVING_THROUGH_STATION:
219  output( new CMessage( this, m_itinerary.get( m_itindex ).m_departuretrack, EMessageType.PASSENGER_TO_PLATFORM_SUBSCRIBE ) );
220  m_state = EPassengerState.ON_PLATFORM_WAITING_FOR_TRAIN;
221  break;
222  case ENTERING_TRAIN:
223  output( new CMessage( this, m_door.id(), EMessageType.PASSENGER_TO_DOOR_FINISHED ) );
224  m_door = null;
225  output( new CMessage( this, m_itinerary.get( m_itindex ).m_trainnumber, EMessageType.PASSENGER_TO_TRAIN_SUBSCRIBE ) );
226  m_state = EPassengerState.ON_TRAIN;
227  break;
228  default:
229  throw new RuntimeException( m_id + " has timed change although in state " + m_state );
230  }
231  return true;
232  }
233 
234  return false;
235  }

237  private boolean handledoormessage()
238  {
239  final List<IMessage> l_yourturn = m_input.get( EMessageType.DOOR_TO_PASSENGER_YOURTURN );
240  if ( !l_yourturn.isEmpty() )
241  {
242  m_door = (IDoor<?>) l_yourturn.get( 0 ).sender();
243  if ( m_state == EPassengerState.IN_ENTRANCE_QUEUE )
244  {
245  output( new CMessage( this, m_itinerary.get( m_itindex ).m_departuretrack, EMessageType.PASSENGER_TO_PLATFORM_UNSUBSCRIBE ) );
246  m_state = EPassengerState.ENTERING_TRAIN;
247  }
248  else if ( m_state == EPassengerState.IN_EXIT_QUEUE )
249  {
250  output( new CMessage( this, m_itinerary.get( m_itindex ).m_trainnumber, EMessageType.PASSENGER_TO_TRAIN_UNSUBSCRIBE ) );
251  m_state = EPassengerState.LEAVING_TRAIN;
252  }
253  else throw new RuntimeException( m_id + " received YOURTURN from " + l_yourturn.get( 0 ).sender() + " although in state " + m_state );
254  // this may have to be changed when the passenger can decide to leave a queue
255  m_dooruse = 0.0;
256  return true;
257  }
258  return false;
259  }

261  private boolean handleplatformmessages()
262  {
263  final List<IMessage> l_trainarrivedatplatform = m_input.get( EMessageType.PLATFORM_TO_PASSENGER_TRAINARRIVED );
264  final List<IMessage> l_traindepartedfromplatform = m_input.get( EMessageType.PLATFORM_TO_PASSENGER_TRAINDEPARTED );
265 
266  if ( !l_trainarrivedatplatform.isEmpty() )
267  {
268  if ( m_state != EPassengerState.ON_PLATFORM_WAITING_FOR_TRAIN )
269  throw new RuntimeException( m_id + " received TRAINARRIVED although in state" + m_state );
270  if ( !m_itinerary.get( m_itindex ).m_trainnumber.equals( l_trainarrivedatplatform.get( 0 ).content()[0] ) ) return false;
271  //for ( final Object l_obj : l_trainarrivedatplatform.get( 0 ).content() )
272  // System.out.println( l_obj.getClass() );
273  final IDoor<?> l_door = (IDoor<?>) ( (Object[]) l_trainarrivedatplatform.get( 0 ).content()[1] )[0];
274  output( new CMessage( this, l_door.id(), EMessageType.PASSENGER_TO_DOOR_ENQUEUE_ENTRANCE ) );
275  m_state = EPassengerState.IN_ENTRANCE_QUEUE;
276  return true;
277  }
278 
279  if ( !l_traindepartedfromplatform.isEmpty() )
280  {
281  if ( m_state == EPassengerState.IN_ENTRANCE_QUEUE )
282  {
283  m_state = EPassengerState.ON_PLATFORM_WAITING_FOR_TRAIN;
284  return true;
285  }
286  else if ( m_state == EPassengerState.ON_PLATFORM_WAITING_FOR_TRAIN ) return false;
287  else throw new RuntimeException( m_id + " received TRAINDEPARTED although in state " + m_state );
288  }
289 
290  return false;
291  }

293  @Override

295  {
296  switch ( m_state )

298 
299  case MOVING_THROUGH_STATION:
300  m_distancewalked += p_elapsed.get( ChronoUnit.SECONDS ) * m_speedatstation;
301  return true;
302  case ENTERING_TRAIN:
303  case LEAVING_TRAIN:
304  m_dooruse += p_elapsed.get( ChronoUnit.SECONDS );
305  return true;
306  default:
307  return false;
