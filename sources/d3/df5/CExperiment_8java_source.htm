220  public IFuzzyValue<Boolean> execute( final boolean p_parallel, @Nonnull final IContext p_context,
221  @Nonnull final List<ITerm> p_argument, @Nonnull final List<ITerm> p_return )
222  {
223  final List<ITerm> l_arguments = CCommon.flatten( p_argument ).collect( Collectors.toList() );
224  if ( l_arguments.size() < 2 )
225  return CFuzzyValue.from( false );
226 
227  final IAgent<?> l_receiver = m_agents.get( l_arguments.get( 0 ).<String>raw() );
228  if ( l_receiver == null )
229  return CFuzzyValue.from( false );
230 
231  final ILiteral l_sender = CAgentTrigger.messagersender( p_context.agent() );
232  l_arguments.stream()
233  .skip( 1 )
234  .map( ITerm::raw )
235  .map( i -> CAgentTrigger.messagesend( i, l_sender ) )
236  .forEach( i -> l_receiver.trigger( i ) );
237 
238  return CFuzzyValue.from( true );
239  }
