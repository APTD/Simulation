156  @Override
157  protected final synchronized Instant determinenextstatechange()
158  {
159  switch ( m_state )
160  {
161  case DRIVING:
162  return m_lastcontinuousupdate.plus(
163  Math.round( ( m_timetable.get( m_ttindex ).m_tracklength - m_positionontrack ) / DRIVING_SPEED ),
164  ChronoUnit.SECONDS
165  );
166  case WAITING_TO_DRIVE:
167  return m_doorsnotclosedlocked.isEmpty() ? m_laststatechange : Instant.MAX;
168  case ARRIVED:
169  if ( m_ttindex + 1 >= m_timetable.size() ) return Instant.MAX;
170  return Collections.max( Arrays.asList( m_timetable.get( m_ttindex ).m_publisheddeparture, m_laststatechange.plus( 30, ChronoUnit.SECONDS ) ) );
171  default:
172  return m_laststatechange.plus( 30, ChronoUnit.SECONDS );
173  }
174  }

176  @Override
177  protected final synchronized boolean updatestate()
178  {
179  final List<IDoor<?>> l_doorsclosedlocked = m_input.get( EMessageType.DOOR_TO_TRAIN_CLOSED_LOCKED ).stream()
180  .map( msg -> (IDoor<?>) msg.sender() ).collect( Collectors.toList() );
181  if ( !l_doorsclosedlocked.isEmpty() )
182  {
183  if ( m_state != ETrainState.WAITING_TO_DRIVE ) throw new RuntimeException( "door locked although not waiting to drive: " + m_id );
184  m_doorsnotclosedlocked.removeAll( l_doorsclosedlocked );
185  m_doorsclosedlocked.addAll( l_doorsclosedlocked );
186  }
187 
188  final List<IMessage> l_subscribingpassengers = m_input.get( EMessageType.PASSENGER_TO_TRAIN_SUBSCRIBE );
189  final List<IMessage> l_unsubscribingpassengers = m_input.get( EMessageType.PASSENGER_TO_TRAIN_UNSUBSCRIBE );
190  if ( !l_subscribingpassengers.isEmpty() || !l_unsubscribingpassengers.isEmpty() )
191  {
192  if ( m_doorsnotclosedlocked.isEmpty() ) throw new RuntimeException( "passengers subscribing/unsubscribing although all doors locked:" + m_id );
193  l_subscribingpassengers.stream().forEach( msg -> m_passengers.add( (IPassenger<?>) msg.sender() ) );
194  l_unsubscribingpassengers.stream().forEach( msg -> m_passengers.remove( msg.sender() ) );
195  }
196 
197  final boolean l_timedchange = !m_nextstatechange.isAfter( m_time.current() );
198  // if ( l_timedchange )
199  // System.out.println( m_id + " - timer transition at " + m_time.current().toString() + " from state " + m_state + " (ttindex = " + m_ttindex + ")" );
200  switch ( m_state )
201  {
202  case ARRIVED:
203  if ( !l_timedchange ) break;
204  System.out.println( m_id + " - departure at " + m_time.current().toString() + " which was planned for "
205  + m_timetable.get( m_ttindex ).m_publisheddeparture );
206  // proceed to next timetable entry
207  m_ttindex++;
208  m_positionontrack = 0.0;
209  m_state = ETrainState.WAITING_TO_DRIVE;
210  m_doorsnotclosedlocked.forEach( d -> output( new CMessage( this, d.id(), EMessageType.TRAIN_TO_DOOR_LOCK, "" ) ) );
211  // debugPrintState();
212  return true;
213  case DRIVING:
214  // @todo: react to "red signal" here
215  if ( !l_timedchange ) break;
216  m_state = ETrainState.ARRIVED;
217  System.out.println( m_id + " - arrival at " + m_time.current().toString() + " which was planned for "
218  + m_timetable.get( m_ttindex ).m_publishedarrival );
219  m_doorsclosedlocked.forEach( d -> output( new CMessage( this, d.id(), EMessageType.TRAIN_TO_DOOR_UNLOCK,
220  m_timetable.get( m_ttindex ).m_stationid,
221  m_timetable.get( m_ttindex ).m_platformid ) ) );
222  m_doorsnotclosedlocked.addAll( m_doorsclosedlocked );
223  m_doorsclosedlocked.clear();
224  m_passengers.forEach( p -> output( new CMessage( this,
225  p.id(),
226  EMessageType.TRAIN_TO_PASSENGER_ARRIVING,
227  m_timetable.get( m_ttindex ).m_stationid,
228  m_timetable.get( m_ttindex ).m_platformid,
229  m_doorsnotclosedlocked.toArray()
230  ) ) );
231  output( new CMessage( this, m_timetable.get( m_ttindex ).m_platformid, EMessageType.TRAIN_TO_PLATFORM_ARRIVING,
232  m_doorsnotclosedlocked.toArray() ) );
233  // debugPrintState();
234  return true;
235  case WAITING_TO_DRIVE:
236  if ( !m_doorsnotclosedlocked.isEmpty() ) break;
237  output( new CMessage( this, m_timetable.get( m_ttindex - 1 ).m_platformid, EMessageType.TRAIN_TO_PLATFORM_DEPARTING ) );
238  m_state = ETrainState.DRIVING;
239  debugPrintState();
240  return true;
241  default:
242  // making checkstyle happy
243  }
244  return false;
245  }
