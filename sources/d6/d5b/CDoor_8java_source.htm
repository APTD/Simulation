131  @Override
132  protected Instant determinenextstatechange()
133  {
134  switch ( m_state )
135  {
136  case CLOSED_LOCKED:
137  case CLOSED_RELEASED:
138  case OPEN_BUSY:
139  case OPEN_BUSY_SHALL_CLOSE:
140  case OPEN_CLOSEABLE:
141  return Instant.MAX;
142  case OPENING:
143  case OPENING_SHALL_CLOSE:
144  return m_lastcontinuousupdate.plus(
145  Math.round( ( m_width - m_openwidth ) / m_openingspeed ),
146  ChronoUnit.SECONDS
147  );
148  case OPEN_FREE:
149  case OPEN_FREE_SHALL_CLOSE:
150  return m_lastcontinuousupdate.plus(
151  Math.round( m_minfreetimetoclose - m_freetime ),
152  ChronoUnit.SECONDS
153  );
154  case CLOSING:
155  case CLOSING_LOCKED:
156  return m_lastcontinuousupdate.plus(
157  Math.round( m_openwidth / m_closingspeed ),
158  ChronoUnit.SECONDS
159  );
160  default:
161  return Instant.MAX;
162  }
163  }

165  @Override
166  protected boolean updatestate()
167  {
168  //debugPrintState();
169  final EDoorState l_oldstate = m_state;
170  final List<IMessage> l_entryrequests = m_input.get( EMessageType.PASSENGER_TO_DOOR_ENQUEUE_ENTRANCE );
171  final List<IMessage> l_exitrequests = m_input.get( EMessageType.PASSENGER_TO_DOOR_ENQUEUE_EXIT );
172  final List<IMessage> l_finished = m_input.get( EMessageType.PASSENGER_TO_DOOR_FINISHED );
173 
174  // enqueuing passengers (if queues are empty, passenger will get immediate access with nextpassengerifpossible() at the end)
175  l_entryrequests.stream().sorted( Comparator.comparing( msg -> msg.sender().id() ) )
176  .forEachOrdered( msg -> m_entryqueue.add( (IPassenger<?>) msg.sender() ) );
177  l_exitrequests.stream().sorted( Comparator.comparing( msg -> msg.sender().id() ) )
178  .forEachOrdered( msg -> m_exitqueue.add( (IPassenger<?>) msg.sender() ) );
179 
180  // locking or unlocking
181  handletrainmessages();
182 
183  // switches state from busy to free, but if queues are not empty, will become busy again with nextpassengerifpossible() at the end
184  if ( l_finished.size() > 1 ) throw new RuntimeException( m_id + " received multiple finished messages" );
185  if ( !l_finished.isEmpty() ) passengerfinished( l_finished.get( 0 ) );
186 
187  // determine from which queue a passenger could be taken next: exit has priority over entry
188  final Queue<IPassenger<?>> l_queue = m_exitqueue.isEmpty() ? m_entryqueue.isEmpty() ? null : m_entryqueue : m_exitqueue;
189 
190  // finish opening/closing process if applicable; switch to closable if door is open and has been unused for m_minfreetimetoclose seconds
191  timedstatechangeifnecessary( l_queue );
192 
193  // if the door is in a permissible state, let the next passenger pass (or open the door first, if closed), if any
194  nextpassengerifpossible( l_queue );
195 
196  //debugPrintState();
197  return l_oldstate != m_state;
198  }

205  private void handletrainmessages()
206  {
207  final List<IMessage> l_lockrequests = m_input.get( EMessageType.TRAIN_TO_DOOR_LOCK );
208  final List<IMessage> l_unlockrequests = m_input.get( EMessageType.TRAIN_TO_DOOR_UNLOCK );
209  final List<IMessage> l_departing = m_input.get( EMessageType.TRAIN_TO_DOOR_DEPARTING );
210 
211  if ( !l_lockrequests.isEmpty() && !l_unlockrequests.isEmpty() )
212  throw new RuntimeException( "door " + m_id + " received lock and unlock request at the same time" );
213 
214  if ( ( !l_lockrequests.isEmpty() || !l_unlockrequests.isEmpty() ) && !l_departing.isEmpty() )
215  throw new RuntimeException( "door " + m_id + " received lock/unlock request and departing message at the same time" );
216 
217  lockifrequested( l_lockrequests );
218 
219  unlockifrequested( l_unlockrequests );
220 
221  if ( l_departing.size() > 1 ) throw new RuntimeException( m_id + " received multiple departing messages" );
222  if ( !l_departing.isEmpty() )
223  {
224  if ( m_state != EDoorState.CLOSED_LOCKED ) throw new RuntimeException( m_id + " received departing message although in state " + m_state );
225  m_entryqueue.forEach( p -> System.out.println( p.id() + " in entry queue when departing" ) );
226  m_exitqueue.forEach( p -> System.out.println( p.id() + " in exit queue when departing" ) );
227  m_entryqueue.clear();
228  m_stationid = null;
229  m_platformid = null;
230  }

234  {

248  case OPEN_FREE_SHALL_CLOSE:

308  private void timedstatechangeifnecessary( final Queue<IPassenger<?>> p_queue )
309  {
310  final boolean l_timedchange = !m_nextstatechange.isAfter( m_time.current() );
311  if ( l_timedchange )
312  {
313  // System.out.println( m_id + " - timer transition at " + m_time.current().toString() + " from state " + m_state );
314  switch ( m_state )
315  {
316  case OPENING:
317  case OPENING_SHALL_CLOSE:
318  m_openwidth = m_width;
319  m_state = m_state == EDoorState.OPENING ? EDoorState.OPEN_FREE : EDoorState.OPEN_FREE_SHALL_CLOSE;
320  m_freetime = 0.0;
321  break;
322  case OPEN_FREE_SHALL_CLOSE:
323  if ( p_queue == null ) m_state = EDoorState.CLOSING_LOCKED;
324  // in case of a passenger enqueueing exactly when the door would start to close, the passenger wins and it does not close
325  break;
326  case OPEN_FREE:
327  m_state = EDoorState.OPEN_CLOSEABLE;
328  break;
329  case CLOSING:
330  m_state = EDoorState.CLOSED_RELEASED;
331  m_openwidth = 0.0;
332  break;
333  case CLOSING_LOCKED:
334  m_state = EDoorState.CLOSED_LOCKED;
335  m_openwidth = 0.0;
336  output( new CMessage( this, m_train, EMessageType.DOOR_TO_TRAIN_CLOSED_LOCKED, "" ) );
337  break;
338  default:
339  // making checkstyle happy
340  }
341  }

344  private void passengerfinished( final IMessage p_message )
345  {
346  switch ( m_state )
347  {
348  case OPEN_BUSY:
349  m_state = EDoorState.OPEN_FREE;
350  break;
351  case OPEN_BUSY_SHALL_CLOSE:
352  m_state = EDoorState.OPEN_FREE_SHALL_CLOSE;
353  break;
354  default:
355  throw new RuntimeException( m_id + " received finished message although not busy in state " + m_state );
356  }
357  final String l_exiting = m_exitqueue.isEmpty() ? null : m_exitqueue.peek().id();
358  final String l_entering = m_entryqueue.isEmpty() ? null : m_entryqueue.peek().id();
359  if ( p_message.sender().id().equals( l_exiting ) ) m_exitqueue.poll();
360  else if ( p_message.sender().id().equals( l_entering ) ) m_entryqueue.poll();
361  else throw new RuntimeException( m_id + " received finished message from " + p_message.sender().id() + " who is not first in either queue" );
362  m_freetime = 0.0;
363  }

365  private void nextpassengerifpossible( final Queue<IPassenger<?>> p_queue )
366  {
367  if ( p_queue != null )
368  {
369  switch ( m_state )
370  {
371  case OPEN_CLOSEABLE:
372  case OPEN_FREE:
373  m_state = EDoorState.OPEN_BUSY;
374  output( new CMessage( this, p_queue.peek().id(), EMessageType.DOOR_TO_PASSENGER_YOURTURN, m_stationid, m_platformid ) );
375  break;
376  case OPEN_FREE_SHALL_CLOSE:
377  m_state = EDoorState.OPEN_BUSY_SHALL_CLOSE;
378  output( new CMessage( this, p_queue.peek().id(), EMessageType.DOOR_TO_PASSENGER_YOURTURN, m_stationid, m_platformid ) );
379  break;
380  case CLOSING:
381  case CLOSED_RELEASED:
382  m_state = EDoorState.OPENING;
383  break;
384  case CLOSING_LOCKED:
385  m_state = EDoorState.OPENING_SHALL_CLOSE;
386  break;
387  default:
388  // making checkstyle happy
389  }

407  case CLOSING:
408  case CLOSING_LOCKED:
409  m_openwidth -= p_elapsed.get( ChronoUnit.SECONDS ) * m_closingspeed;
410  return true;
411  default:
412  // making checkstyle happy
413  }
414  return false;
415  }
