71  public DoubleMatrix1D position( final DoubleMatrix1D p_position, final DoubleMatrix1D p_goalposition, final int p_speed )
72  {
73  // calculate the stright line by: current position + l * (goal position - current position)
74  // normalize direction and rotate the normalized vector based on the direction
75  // calculate the target position based by: current position + speed * rotate( normalize( goal position - current position ) )
76  final DoubleMatrix1D l_view = new DenseDoubleMatrix1D( p_goalposition.toArray() );
77  return CMath.ALGEBRA.mult(
78  m_rotation,
79  l_view
80  .assign( p_position, Functions.minus )
81  .assign( Functions.div( Math.sqrt( Algebra.DEFAULT.norm2( l_view ) ) ) )
82  )
83  .assign( Functions.mult( p_speed ) )
84  .assign( p_position, Functions.plus )
85  .assign( Math::round );
86  }
