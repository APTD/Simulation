130  private static void execute( final CommandLine p_cli )
131  {
132  final double l_passengerspeedmin = Double.parseDouble( p_cli.getOptionValue( "passengerspeedmin", "1.5" ) );
133  final double l_passengerspeedmax = Double.parseDouble( p_cli.getOptionValue( "passengerspeedmax", "1.5" ) );
134 
135  // load configuration
136  CConfiguration.INSTANCE.loadfile( p_cli.getOptionValue( "config", "" ) );
137 
138  // execute experiments in batch processing and starts http server
139  new Thread( () -> datamodel( p_cli )
140  .map( i -> i.getLeft().model().get(
141 
142  EFactory.from( CConfiguration.INSTANCE.getOrDefault( "local", "runtime", "type" ) ).factory(),
143 
144  i.getRight(),
145 
146  p_cli.hasOption( "iteration" )
147  ? Long.parseLong( p_cli.getOptionValue( "iteration" ) )
148  : (long) CConfiguration.INSTANCE.getOrDefault( 0, "default", "iteration" ),
149 
150  !p_cli.hasOption( "sequential" ) && CConfiguration.INSTANCE.<Boolean>getOrDefault( true, "runtime", "parallel" ),
151 
152  p_cli.hasOption( "timemodel" )
153  ? p_cli.getOptionValue( "timemodel" )
154  : "step",
155 
156  () ->
157  {
158  if ( l_passengerspeedmax <= l_passengerspeedmin ) return new ConstantRealDistribution( l_passengerspeedmin );
159  final RandomGenerator l_randomgenerator = new JDKRandomGenerator();
160  l_randomgenerator.setSeed( Long.parseLong( p_cli.getOptionValue( "passengerspeedseed", "1" ) ) );
161  return new UniformRealDistribution( l_randomgenerator, l_passengerspeedmin, l_passengerspeedmax );
162  }
163 
164  ) )
165  .forEach( i -> ERuntime.LOCAL.get().execute( i ) )
166  ).start();
167 
168  // start http server if possible
169  if ( p_cli.hasOption( "interactive" ) ) CHTTPServer.execute();
170  }

179  private static Stream<Pair<EDataModel, String>> datamodel( final CommandLine p_options )
180  {
181  final List<String> l_instances = Arrays.stream( p_options.getOptionValue( "scenario" ).split( "," ) )
182  .map( String::trim )
183  .filter( i -> !i.isEmpty() )
184  .collect( Collectors.toList() );
185 
186  final List<String> l_types = Arrays.stream( p_options.getOptionValue( "scenariotype", "" ).split( "," ) )
187  .map( String::trim )
188  .filter( i -> !i.isEmpty() )
189  .collect( Collectors.toList() );
190 
191  return StreamUtils.zip(
192  l_instances.stream(),
193  Stream.concat(
194  l_types.stream(),
195  IntStream.range( 0, l_instances.size() - l_types.size() )
196  .mapToObj( i -> CConfiguration.INSTANCE.getOrDefault( "xml", "default", "datamodel" ) )
197  ),
198  ( i, j ) -> new ImmutablePair<>( EDataModel.from( j ), i )
199  );
200  }
